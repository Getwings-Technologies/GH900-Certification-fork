# Title: Real-time Python Script Sharing Example

# Description:
# This snippet demonstrates a simple Python script that prints system time every second.
# Share this gist with teammates for quick reference or collaboration.

import time

def print_real_time():
    try:
        while True:
            print("Current Time:", time.strftime("%Y-%m-%d %H:%M:%S"))
            time.sleep(1)
    except KeyboardInterrupt:
        print("\nStopped real-time clock.")

if __name__ == "__main__":
    print_real_time()
